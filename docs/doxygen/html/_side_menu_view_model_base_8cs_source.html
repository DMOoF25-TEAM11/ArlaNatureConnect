164 protected virtual void OnNavigate(object? parameter)
165 {
166 try
167 {
168 ArgumentNullException.ThrowIfNull(parameter);
169 if (_navigationViewModel == null)
170 throw new InvalidOperationException("No host navigation view-model set.");
171
172
173 // If parameter is a UserControl, set it directly as the host page content
174 if (parameter is UserControl uc && _navigationViewModel is NavigationViewModelBase hostVm)
175 {
176 try { uc.DataContext = hostVm; } catch { }
177 hostVm.GetType().GetProperty("CurrentContent")?.SetValue(hostVm, uc);
178 return;
179 }
180
181 // If parameter is a factory that returns a control or element, invoke and assign
182 if (parameter is Func<object?> factory && _navigationViewModel is NavigationViewModelBase hostVm2)
183 {
184 object? obj = null;
185 try { obj = factory(); } catch { obj = null; }
186 if (obj is FrameworkElement fe)
187 {
188 try { fe.DataContext = hostVm2; } catch { }
189 }
190 if (obj is UserControl userControl)
191 {
192 hostVm2.GetType().GetProperty("CurrentContent")?.SetValue(hostVm2, userControl);
193 }
194 return;
195 }
196
197 // Otherwise try forwarding to host navigation command if available
198 if (_navigationViewModel is NavigationViewModelBase hostVm3)
199 {
200 RelayCommand<object>? hostCmd = hostVm3.NavigationCommand;
201 if (hostCmd != null && hostCmd.CanExecute(parameter))
202 {
203 hostCmd.Execute(parameter);
204 return;
205 }
206 }
207 }
208 catch (Exception ex)
209 {
210 throw new InvalidOperationException("OnNavigate failed", ex);
211 }
212
213 // No host available or unable to set content: remain a no-op.
214 }
