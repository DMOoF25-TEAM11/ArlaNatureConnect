90 public async Task<(bool IsValid, string? ErrorMessage)> ValidateConnectionStringWithRetryAsync(string connectionString)
91 {
92 if (string.IsNullOrWhiteSpace(connectionString))
93 return (false, "Connection string is empty.");
94
95 SqlConnectionStringBuilder builder = new SqlConnectionStringBuilder(connectionString);
96 if (string.IsNullOrWhiteSpace(builder.DataSource))
97 return (false, "Connection string is missing a server/data source. Please enter a valid SQL Server instance.");
98
99 // Ensure a short timeout for the UI test
100 int originalTimeout = builder.ConnectTimeout;
101 if (originalTimeout <= 0 || originalTimeout > 10)
102 {
103 builder.ConnectTimeout = 5;
104 }
105
106 const int maxAttempts = 3;
107
108 for (int attempt = 1; attempt <= maxAttempts; attempt++)
109 {
110 try
111 {
112 using SqlConnection conn = new SqlConnection(builder.ConnectionString);
113 await conn.OpenAsync();
114 await conn.CloseAsync();
115 _statusInfoServices?.HasDbConnection = true;
116 return (true, null);
117 }
118 catch (SqlException ex)
119 {
120 _statusInfoServices?.HasDbConnection = false;
121 if (attempt == maxAttempts)
122 {
123 return (false, $"Unable to connect to SQL Server '{builder.DataSource}': {ex.Message}");
124 }
125 await Task.Delay(500);
126 }
127 catch (System.ComponentModel.Win32Exception ex)
128 {
129 if (attempt == maxAttempts)
130 {
131 return (false, $"Network error while attempting to connect to SQL Server '{builder.DataSource}': {ex.Message}");
132 }
133 await Task.Delay(500);
134 }
135 catch (Exception ex)
136 {
137 return (false, $"Unexpected error while validating connection: {ex.Message}");
138 }
139 }
140
141 return (false, "Unknown error validating connection string.");
142 }
