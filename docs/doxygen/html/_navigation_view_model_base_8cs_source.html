248 {
249 try
250 {
251 MainWindow? mainWindow = App.HostInstance?.Services.GetService<MainWindow>();
252 if (mainWindow?.Content is not FrameworkElement root) return;
253 Panel? sideMenuPanel = FindSideMenuPanel(root);
254 if (sideMenuPanel == null) return;
255
256 //if (_addedSideMenuControl is FrameworkElement added && sideMenuPanel.Children.Contains(added))

262
263 _previousSideMenuChildren = [.. sideMenuPanel.Children.Cast<UIElement>()];
264 sideMenuPanel.Children.Clear();
265
266 object? resolvedVm = null;
267 UserControl? control = null;
268
269 // Prefer explicit type if provided; create a scope for scoped services so DI lifetime is correct
270 if (SideMenuViewModelType != null && App.HostInstance != null)
271 {
272 try
273 {
274 // dispose previous scope if any
275 _sideMenuScope?.Dispose();
276 _sideMenuScope = App.HostInstance.Services.CreateScope();
277 resolvedVm = _sideMenuScope.ServiceProvider.GetService(SideMenuViewModelType);
278 }
279 catch { resolvedVm = null; }
280 }
281 else
282 {
283 // Fallback convention: replace Views.Controls.SideMenu namespace with ViewModels.Controls.SideMenu and append ViewModel
284 if (control != null)
285 {
286 Type controlType = control.GetType();
287 string? ns = controlType.Namespace;
288 if (!string.IsNullOrWhiteSpace(ns) && ns.Contains("Views.Controls.SideMenu"))
289 {
290 string vmNs = ns.Replace("Views.Controls.SideMenu", "ViewModels.Controls.SideMenu");
291 string vmFull = vmNs + "." + controlType.Name + "ViewModel";
292 Type? vmType = Type.GetType(vmFull);
293 if (vmType != null)
294 {
295 try
296 {
297 _sideMenuScope?.Dispose();
298 _sideMenuScope = App.HostInstance?.Services.CreateScope();
299 resolvedVm = _sideMenuScope?.ServiceProvider.GetService(vmType);
300 }
301 catch { resolvedVm = null; }
302 }
303 }
304 }
305 }
306
307
308 // 1) If explicit view type provided by VM, try to resolve or instantiate it
309 if (SideMenuControlType != null && typeof(UserControl).IsAssignableFrom(SideMenuControlType))
310 {
311 control = App.HostInstance?.Services.GetService(SideMenuControlType) as UserControl
312 ?? Activator.CreateInstance(SideMenuControlType) as UserControl;
313 }
314
315 // 2) If VM type was resolved but no control yet, try convention-based view lookup
316 if (control == null && resolvedVm != null)
317 {
318 Type vmType = resolvedVm.GetType();
319 // Example convention: ViewModels.Controls.SideMenu.XxxViewModel -> Views.Controls.SideMenu.XxxUC
320 string vmNs = vmType.Namespace ?? string.Empty;
321 string viewNs = vmNs.Replace("ViewModels", "Views");
322 string viewName = vmType.Name.Replace("ViewModel", "UC"); // adjust if project uses different suffix
323 string fullViewType = $"{viewNs}.{viewName}";
324 Type? viewType = Type.GetType(fullViewType);
325 if (viewType != null && typeof(UserControl).IsAssignableFrom(viewType))
326 {
327 control = App.HostInstance?.Services.GetService(viewType) as UserControl
328 ?? Activator.CreateInstance(viewType) as UserControl;
329 }
330 }
331
332 // 3) Final safety: if still null, bail out instead of dereferencing
333 if (control == null)
334 {
335 // log or handle the missing view; restore previous children
336 if (_previousSideMenuChildren != null)
337 {
338 foreach (UIElement child in _previousSideMenuChildren) sideMenuPanel.Children.Add(child);
339 _previousSideMenuChildren = null;
340 }
341 return;
342 }
343
344 // now safe to set DataContext and add the control
345 if (resolvedVm != null)
346 {
347 // Preferred typed call if using the concrete base type
348 if (resolvedVm is SideMenuViewModelBase sideVm)
349 {
350 sideVm.SetHostPageViewModel(this);
351 }
352 else
353 {
354 // Fallback: try to call a SetHostPageViewModel method if present
355 MethodInfo? method = resolvedVm.GetType().GetMethod("SetHostPageViewModel", [typeof(object)]);
356 method?.Invoke(resolvedVm, [this]);
357 }
358 }
359
360 control.DataContext = resolvedVm ?? this;
361 _addedSideMenuControl = control;
362 sideMenuPanel.Children.Add(control);
363 }
364 catch { }
365 }

415 private static Panel? FindSideMenuPanel(FrameworkElement root)
416 {
417 if (root == null) return null;
418
419 // Try straightforward FindName first
420 if (root.FindName("SideMenu") is Panel p)
421 return p;
422
423 // Fallback: traverse visual tree
424 try
425 {
426 int count = VisualTreeHelper.GetChildrenCount(root);
427 for (int i = 0; i < count; i++)
428 {
429 DependencyObject child = VisualTreeHelper.GetChild(root, i);
430 if (child is Panel panel && panel.Name == "SideMenu") return panel;
431 if (child is FrameworkElement fe)
432 {
433 Panel? found = FindSideMenuPanel(fe);
434 if (found != null) return found;
435 }
436 }
437 }
438 catch { }
439
440 return null;
