53 public async Task<NatureCheckCaseAssignmentContext> LoadAssignmentContextAsync(CancellationToken cancellationToken = default)
54 {
55 // Materialize all results to avoid concurrency issues with shared DbContext
56 List<Farm> farms = (await _farmRepository.GetAllAsync(cancellationToken).ConfigureAwait(false)).ToList();
57 List<Person> allPersons = (await _personRepository.GetAllAsync(cancellationToken).ConfigureAwait(false)).ToList();
58 List<Person> consultants = (await _personRepository.GetPersonsByRoleAsync(RoleName.Consultant.ToString(), cancellationToken).ConfigureAwait(false)).ToList();
59 List<Address> addresses = (await _addressRepository.GetAllAsync(cancellationToken).ConfigureAwait(false)).ToList();
60 IReadOnlyList<NatureCheckCase> activeCases = await _natureCheckCaseRepository.GetActiveCasesAsync(cancellationToken).ConfigureAwait(false);
61
62 Dictionary<Guid, Person> personsById = allPersons.ToDictionary(p => p.Id, p => p);
63 Dictionary<Guid, Address> addressesById = addresses.ToDictionary(a => a.Id, a => a);
64 // GetActiveCasesAsync already filters for Assigned and InProgress, so no need to filter again
65 HashSet<Guid> activeFarmIds = activeCases
66 .Select(c => c.FarmId)
67 .ToHashSet();
68
69 Dictionary<Guid, NatureCheckCase> activeCasesByFarm = activeCases
70 .GroupBy(c => c.FarmId)
71 .Select(g => g.OrderByDescending(c => c.AssignedAt ?? c.CreatedAt).First())
72 .ToDictionary(c => c.FarmId, c => c);
73
74 List<FarmAssignmentOverviewDto> overview = farms
75 .Select(f => CreateFarmOverview(f, personsById, addressesById, activeFarmIds.Contains(f.Id), activeCasesByFarm))
76 .OrderBy(f => f.FarmName)
77 .ToList();
78
79 List<Person> sortedConsultants = consultants
80 .OrderBy(c => c.FirstName)
81 .ThenBy(c => c.LastName)
82 .ToList();
83
84 return new NatureCheckCaseAssignmentContext(overview, sortedConsultants);
85 }

182 public async Task<Farm> SaveFarmAsync(FarmRegistrationRequest request, CancellationToken cancellationToken = default)
183 {
184 ArgumentNullException.ThrowIfNull(request);
185
186 ValidateFarmRegistration(request);
187
188 if (request.FarmId.HasValue)
189 {
190 Farm? existingFarm = await _farmRepository.GetByIdAsync(request.FarmId.Value, cancellationToken).ConfigureAwait(false) ?? throw new InvalidOperationException("Kunne ikke finde gården til redigering.");
191 if (existingFarm.AddressId != Guid.Empty)
192 {
193 Address? farmAddress = await _addressRepository.GetByIdAsync(existingFarm.AddressId, cancellationToken).ConfigureAwait(false);
194 if (farmAddress != null)
195 {
196 farmAddress.Street = request.Street;
197 farmAddress.City = request.City;
198 farmAddress.PostalCode = request.PostalCode;
199 farmAddress.Country = request.Country;
200 await _addressRepository.UpdateAsync(farmAddress, cancellationToken).ConfigureAwait(false);
201 }
202 }
203
204 if (existingFarm.OwnerId != Guid.Empty)
205 {
206 Person? farmer = await _personRepository.GetByIdAsync(existingFarm.OwnerId, cancellationToken).ConfigureAwait(false);
207 if (farmer != null)
208 {
209 farmer.FirstName = request.OwnerFirstName;
210 farmer.LastName = request.OwnerLastName;
211 farmer.Email = request.OwnerEmail;
212 await _personRepository.UpdateAsync(farmer, cancellationToken).ConfigureAwait(false);
213 }
214 }
215
216 existingFarm.Name = request.FarmName;
217 existingFarm.CVR = request.Cvr;
218 await _farmRepository.UpdateAsync(existingFarm, cancellationToken).ConfigureAwait(false);
219
220 return existingFarm;
221 }
222 else
223 {
224 // Validate that CVR doesn't already exist (if provided)
225 if (!string.IsNullOrWhiteSpace(request.Cvr))
226 {
227 Farm? existingFarm = await _farmRepository.GetByCvrAsync(request.Cvr, cancellationToken).ConfigureAwait(false);
228 if (existingFarm != null)
229 {
230 throw new InvalidOperationException($"En gård med CVR-nummer '{request.Cvr}' findes allerede i systemet. Vælg et andet CVR-nummer.");
231 }
232 }
233
234 // Check if person with this email already exists
235 Person? existingPerson = await _personRepository.GetByEmailAsync(request.OwnerEmail, cancellationToken).ConfigureAwait(false);
236 Person farmer;
237 ArlaNatureConnect.Domain.Entities.Role farmerRole = await EnsureRoleAsync(RoleName.Farmer.ToString(), cancellationToken).ConfigureAwait(false);
238
239 if (existingPerson != null)
240 {
241 // Person exists - verify they have Farmer role
242 Role? existingRole = await _roleRepository.GetByIdAsync(existingPerson.RoleId, cancellationToken).ConfigureAwait(false);
243 if (existingRole == null || existingRole.Name != RoleName.Farmer.ToString())
244 {
245 throw new InvalidOperationException($"En person med e-mail '{request.OwnerEmail}' findes allerede i systemet, men har ikke rollen 'Farmer'. En landmand kan kun have flere gårde hvis de har Farmer-rollen.");
246 }
247
248 // Use existing person - they can have multiple farms
249 farmer = existingPerson;
250 }
251 else
252 {
253 // Create new person address (if provided)
254 Guid? personAddressId = null;
255 if (!string.IsNullOrWhiteSpace(request.Street) || !string.IsNullOrWhiteSpace(request.City))
256 {
257 Address personAddress = new()
258 {
259 Id = Guid.NewGuid(),
260 Street = request.Street,
261 City = request.City,
262 PostalCode = request.PostalCode,
263 Country = request.Country
264 };
265 await _addressRepository.AddAsync(personAddress, cancellationToken).ConfigureAwait(false);
266 personAddressId = personAddress.Id;
267 }
268
269 // Create new person
270 farmer = new()
271 {
272 Id = Guid.NewGuid(),
273 RoleId = farmerRole.Id,
274 AddressId = personAddressId ?? Guid.Empty,
275 FirstName = request.OwnerFirstName,
276 LastName = request.OwnerLastName,
277 Email = request.OwnerEmail,
278 IsActive = request.OwnerIsActive
279 };
280 await _personRepository.AddAsync(farmer, cancellationToken).ConfigureAwait(false);
281 }
282
283 // Create farm address
284 Address farmAddress = new()
285 {
286 Id = Guid.NewGuid(),
287 Street = request.Street,
288 City = request.City,
289 PostalCode = request.PostalCode,
290 Country = request.Country
291 };
292 await _addressRepository.AddAsync(farmAddress, cancellationToken).ConfigureAwait(false);
293
294 // Create new farm linked to the person (existing or new)
295 Farm farm = new()
296 {
297 Id = Guid.NewGuid(),
298 Name = request.FarmName,
299 CVR = request.Cvr,
300 OwnerId = farmer.Id,
301 AddressId = farmAddress.Id
302 };
303 await _farmRepository.AddAsync(farm, cancellationToken).ConfigureAwait(false);
304
305 return farm;
306 }

326 public async Task<IReadOnlyList<ConsultantNotificationDto>> GetNotificationsForConsultantAsync(Guid consultantId, CancellationToken cancellationToken = default)
327 {
328 IReadOnlyList<NatureCheckCase> assignedCases = await _natureCheckCaseRepository.GetAssignedCasesForConsultantAsync(consultantId, cancellationToken).ConfigureAwait(false);
329
330 if (assignedCases.Count == 0)
331 {
332 return Array.Empty<ConsultantNotificationDto>();
333 }
334
335 // Get all farm IDs
336 HashSet<Guid> farmIds = assignedCases.Select(c => c.FarmId).ToHashSet();
337
338 // Load all farms in one query
339 List<Farm> farms = (await _farmRepository.GetAllAsync(cancellationToken).ConfigureAwait(false)).Where(f => farmIds.Contains(f.Id)).ToList();
340
341 Dictionary<Guid, Farm> farmsById = farms.ToDictionary(f => f.Id, f => f);
342
343 List<ConsultantNotificationDto> notifications = new();
344 foreach (NatureCheckCase caseEntity in assignedCases)
345 {
346 if (!farmsById.TryGetValue(caseEntity.FarmId, out Farm? farm))
347 {
348 continue; // Skip if farm not found
349 }
350
351 notifications.Add(new ConsultantNotificationDto
352 {
353 CaseId = caseEntity.Id,
354 FarmId = caseEntity.FarmId,
355 FarmName = farm.Name,
356 AssignedAt = caseEntity.AssignedAt ?? caseEntity.CreatedAt,
357 Priority = caseEntity.Priority,
358 Notes = caseEntity.Notes
359 });
360 }
361
362 return notifications.OrderByDescending(n => n.AssignedAt).ToList();
363 }

369 Dictionary<Guid, Person> personsById,
370 Dictionary<Guid, Address> addressesById,

393 {
394 consultantId = activeCase.ConsultantId;
395 priority = activeCase.Priority;
396 notes = activeCase.Notes;
397

399 {
400 consultantFirstName = consultant.FirstName;
401 consultantLastName = consultant.LastName;
402 }
403 }
404
405 return new FarmAssignmentOverviewDto
406 {
407 FarmId = farm.Id,
408 FarmName = farm.Name,
409 Cvr = farm.CVR,
410 OwnerFirstName = owner?.FirstName ?? string.Empty,
411 OwnerLastName = owner?.LastName ?? string.Empty,
412 OwnerEmail = owner?.Email ?? string.Empty,
413 Street = address?.Street ?? string.Empty,
414 City = address?.City ?? string.Empty,
415 PostalCode = address?.PostalCode ?? string.Empty,
416 Country = address?.Country ?? string.Empty,
417 HasActiveCase = hasActiveCase,
418 AssignedConsultantFirstName = consultantFirstName,
419 AssignedConsultantLastName = consultantLastName,
420 AssignedConsultantId = consultantId,
421 Priority = priority,
422 Notes = notes
423 };
