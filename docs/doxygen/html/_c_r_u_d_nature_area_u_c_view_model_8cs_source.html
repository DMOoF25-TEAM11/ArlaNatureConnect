202 private static double CalculatePolygonAreaKm2(IEnumerable<NatureAreaCoordinate> coordinates)
203 {
204 // Uses the spherical excess formula for area on a sphere (Earth)
205 // Assumes coordinates are ordered and form a closed polygon
206 const double EarthRadius = 6371.0; // km
207 List<NatureAreaCoordinate>? coords = coordinates?.OrderBy(c => c.OrderIndex).ToList();
208 if (coords == null || coords.Count < 3) return 0.0;
209 double area = 0.0;
210 int n = coords.Count;
211 for (int i = 0; i < n; i++)
212 {
213 NatureAreaCoordinate p1 = coords[i];
214 NatureAreaCoordinate p2 = coords[(i + 1) % n];
215 double lat1 = p1.Latitude * Math.PI / 180.0;
216 double lon1 = p1.Longitude * Math.PI / 180.0;
217 double lat2 = p2.Latitude * Math.PI / 180.0;
218 double lon2 = p2.Longitude * Math.PI / 180.0;
219 area += (lon2 - lon1) * (2 + Math.Sin(lat1) + Math.Sin(lat2));
220 }
221 area = area * EarthRadius * EarthRadius / 2.0;
222 return Math.Abs(area); // in km^2
223 }
