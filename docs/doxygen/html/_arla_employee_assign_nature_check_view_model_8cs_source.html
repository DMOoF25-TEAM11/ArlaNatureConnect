305 private async Task AssignNatureCheckCaseAsync()
306 {
307 if (!CanAssignCase())
308 {
309 return;
310 }
311
312 // Try to acquire the assignment semaphore without waiting. If another assignment is in progress,
313 // exit early to avoid multiple concurrent assignments (command may be invoked directly in tests).
314 if (!_assignSemaphore.Wait(0))
315 {
316 return;
317 }
318
319 try
320 {
321 await ExecuteWithBusyStateAsync(async () =>
322 {
323 NatureCheckCaseAssignmentRequest request = new()
324 {
325 FarmId = SelectedFarm!.FarmId,
326 ConsultantId = SelectedConsultant!.Id,
327 AssignedByPersonId = AssignedByPersonId ?? Guid.Empty,
328 Notes = AssignmentNotes,
329 Priority = ToEnglish(SelectedPriority),
330 AllowDuplicateActiveCase = false
331 };
332
333 // Check if farm already has an active case - update instead of create
334 if (SelectedFarm.HasActiveCase)
335 {
336 await _natureCheckCaseService.UpdateCaseAsync(SelectedFarm.FarmId, request);
337 _appMessageService.AddInfoMessage($"Natur Check opgave er opdateret for {SelectedFarm.FarmName}.");
338 }
339 else
340 {
341 await _natureCheckCaseService.AssignCaseAsync(request);
342 _appMessageService.AddInfoMessage($"Natur Check opgave er oprettet for {SelectedFarm.FarmName}.");
343 }
344
345 AssignmentNotes = string.Empty;
346 SelectedPriority = null;
347 await EnsureAssignmentDataAsync(forceReload: true);
348 });
349 }
350 finally
351 {
352 _assignSemaphore.Release();
353 }
354 }

365 private async Task EnsureAssignmentDataAsync(bool forceReload = false)
366 {
367 if (_assignmentLoaded && !forceReload)
368 {
369 return;
370 }
371
372 try
373 {
374 using IDisposable loading = _statusInfoServices.BeginLoadingOrSaving();
375
376 NatureCheckCaseAssignmentContext context = await _natureCheckCaseService.LoadAssignmentContextAsync();
377 if (context == null)
378 {
379 // Defensive: service returned null (mock or faulty implementation). Treat as failure to load.
380 _appMessageService.AddErrorMessage("Kunne ikke hente data: context was null");
381 _assignmentLoaded = false;
382 return;
383 }
384 UpdateFarms(context.Farms);
385 UpdateConsultants(context.Consultants);
386
387 _assignmentLoaded = true;
388 }
389 catch (Exception ex)
390 {
391 _appMessageService.AddErrorMessage($"Kunne ikke hente data: {ex.Message}");
392 _assignmentLoaded = false; // Reset flag so we can retry
393 }

505 {
506 using IDisposable loading = _statusInfoServices.BeginLoadingOrSaving();
507
508 try

531 private static string ExtractDatabaseErrorMessage(Microsoft.EntityFrameworkCore.DbUpdateException dbEx)
532 {
533 // Check inner exception for SQL Server specific errors
534 if (dbEx.InnerException is Microsoft.Data.SqlClient.SqlException sqlEx)
535 {
536 // Check for unique constraint violations
537 if (sqlEx.Number == 2627) // Unique constraint violation
538 {
539 string message = sqlEx.Message;
540
541 // Check if it's an email constraint
542 if (message.Contains("UQ_Persons_Email", StringComparison.OrdinalIgnoreCase) ||
543 message.Contains("Email", StringComparison.OrdinalIgnoreCase))
544 {
545 return "En person med denne e-mail findes allerede i systemet. Vælg en anden e-mail.";
546 }
547
548 // Check if it's a CVR constraint
549 if (message.Contains("UQ_Farms_CVR", StringComparison.OrdinalIgnoreCase) ||
550 message.Contains("CVR", StringComparison.OrdinalIgnoreCase))
551 {
552 return "En gård med dette CVR-nummer findes allerede i systemet. Vælg et andet CVR-nummer.";
553 }
554
555 return "En post med disse oplysninger findes allerede i systemet. Tjek at e-mail og CVR er unikke.";
556 }
557
558 // Check for foreign key violations
559 if (sqlEx.Number == 547) // Foreign key constraint violation
560 {
561 return "Kan ikke gemme: Der er en reference til data der ikke findes. Kontakt systemadministratoren.";
562 }
563
564 // Return SQL error message if it's meaningful
565 if (!string.IsNullOrWhiteSpace(sqlEx.Message))
566 {
567 return sqlEx.Message;
568 }
569 }
570
571 // Fallback to original exception message
572 return dbEx.Message;
573 }

610 private void ApplyFilters()
611 {
612 IEnumerable<AssignableFarmViewModel> query = _allFarms.AsEnumerable();
613
614 if (!string.IsNullOrWhiteSpace(FarmSearchText))
615 {
616 string search = FarmSearchText.Trim().ToLowerInvariant();
617 query = query.Where(f => f.SearchContent.Contains(search));
618 }
619
620 query = _currentStatusFilter switch
621 {
622 FarmStatusFilter.Assigned => query.Where(f => f.HasActiveCase),
623 FarmStatusFilter.Unassigned => query.Where(f => !f.HasActiveCase),
624 _ => query
625 };
626
627 SyncCollection(FilteredFarms, query);
628 if (SelectedFarm != null && !FilteredFarms.Contains(SelectedFarm))
629 {
630 SelectedFarm = FilteredFarms.FirstOrDefault();
631 }
