280 private void ParseConnectionString(string? connectionString)
281 {
282 _suppressUpdate = true;
283 try
284 {
285 // Reset
286 ResetFields(updateConnectionString: false);
287
288 if (string.IsNullOrWhiteSpace(connectionString)) return;
289
290 Dictionary<string, string> entries = connectionString
291 .Split(';', StringSplitOptions.RemoveEmptyEntries)
292 .Select(p =>
293 {
294 int idx = p.IndexOf('=');
295 if (idx < 0) return new KeyValuePair<string, string>(p.Trim(), string.Empty);
296 string key = p[..idx].Trim();
297 string val = p[(idx + 1)..].Trim();
298 return new KeyValuePair<string, string>(key, val);
299 })
300 .ToDictionary(kv => kv.Key, kv => kv.Value, StringComparer.OrdinalIgnoreCase);
301
302 if (entries.TryGetValue("Server", out string? server) || entries.TryGetValue("Data Source", out server))
303 {
304 ServerName = server;
305 }
306
307 if (entries.TryGetValue("Database", out string? database) || entries.TryGetValue("Initial Catalog", out database))
308 {
309 DatabaseName = database;
310 }
311
312 if (entries.TryGetValue("Integrated Security", out string? integ) || entries.TryGetValue("Trusted_Connection", out integ))
313 {
314 IntegratedSecurity = string.Equals(integ, "True", StringComparison.OrdinalIgnoreCase) || string.Equals(integ, "SSPI", StringComparison.OrdinalIgnoreCase);
315 }
316 else
317 {
318 IntegratedSecurity = false;
319 }
320
321 if (entries.TryGetValue("User ID", out string? user))
322 {
323 UserName = user;
324 }
325

327 {
328 Password = pwd;
329 }
330
