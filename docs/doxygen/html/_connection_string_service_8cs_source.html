24 public async Task<string?> ReadAsync()
25 {
26 if (!File.Exists(_filePath)) return null;
27
28 // Try to open and read the file directly with sharing allowed.
29 // If the file is temporarily locked, retry until timeout.
30 TimeSpan timeout = TimeSpan.FromSeconds(5);
31 using CancellationTokenSource cts = new CancellationTokenSource(timeout);
32 CancellationToken ct = cts.Token;
33 TimeSpan delay = TimeSpan.FromMilliseconds(100);
34
35 while (true)
36 {
37 ct.ThrowIfCancellationRequested();
38 try
39 {
40 // Allow others to read/write while we read to avoid blocking writers.
41 using FileStream fs = new(_filePath, FileMode.Open, FileAccess.Read, FileShare.ReadWrite, 4096, useAsync: true);
42 if (fs.Length == 0) return null;
43 byte[] encrypted = new byte[fs.Length];
44 int offset = 0;
45 while (offset < encrypted.Length)
46 {
47 int read = await fs.ReadAsync(encrypted.AsMemory(offset, encrypted.Length - offset)).ConfigureAwait(false);
48 if (read == 0) break;
49 offset += read;
50 }
51 return await DecryptAsync(encrypted).ConfigureAwait(false);
52 }
53 catch (OperationCanceledException)
54 {
55 throw new IOException("Timed out waiting to read the connection file.");
56 }
57 catch (IOException)
58 {
59 // File currently locked by a writer -> wait and retry
60 try
61 {
62 await Task.Delay(delay, ct).ConfigureAwait(false);
63 }
64 catch (OperationCanceledException)
65 {
66 throw new IOException("Timed out waiting to read the connection file.");
67 }
68 }
69 catch
70 {
71 throw new InvalidOperationException("Failed to read or decrypt the connection string.");
72 }
73 }
74 }

162 private static Task<string?> DecryptNonWindowsOsAsync(byte[] encryptedData)
163 {
164 try
165 {
166 Debug.WriteLine("Using AES decryption for connection string on non-Windows OS.");
167 byte[] key = DeriveKey();
168 using Aes aes = Aes.Create();
169 aes.Key = key;
170 aes.Mode = CipherMode.CBC;
171 aes.Padding = PaddingMode.PKCS7;
172
173 // IV is prepended
174 if (encryptedData.Length < aes.BlockSize / 8) return Task.FromResult<string?>(null);
175 byte[] iv = new byte[aes.BlockSize / 8];
176 Array.Copy(encryptedData, 0, iv, 0, iv.Length);
177 byte[] cipher = new byte[encryptedData.Length - iv.Length];
178 Array.Copy(encryptedData, iv.Length, cipher, 0, cipher.Length);
179 aes.IV = iv;
180
181 using MemoryStream ms = new();
182 using (CryptoStream cs = new(ms, aes.CreateDecryptor(), CryptoStreamMode.Write))
183 {
184 cs.Write(cipher, 0, cipher.Length);
185 cs.FlushFinalBlock();
186 }
187
188 return Task.FromResult<string?>(Encoding.UTF8.GetString(ms.ToArray()));
189 }
190 catch
191 {
192 return Task.FromResult<string?>(null);
193 }

197 private static byte[] DeriveKey()
198 {
199 // Derive a256-bit key from user and machine information and application base path.
200 // This makes the encrypted data specific to the current user on the current machine.
201 StringBuilder sb = new();
202 sb.Append(Environment.UserName ?? string.Empty);
203 sb.Append('|');
204 sb.Append(Environment.MachineName ?? string.Empty);
205 sb.Append('|');
206 sb.Append(AppDomain.CurrentDomain?.BaseDirectory ?? string.Empty);
207
208 // Use SHA256 to produce a32-byte key. This is deterministic but tied to user+machine+app path.
209 return SHA256.HashData(Encoding.UTF8.GetBytes(sb.ToString()));
210 }

212 public async Task<bool> CanConnectionStringConnect(string connectionString)
213 {
214 if (string.IsNullOrWhiteSpace(connectionString))
215 {
216 Debug.WriteLine("Connection string is null or empty.");
217 return false;
218 }
219
220 // Quick validation: ensure a data source is present and that we can open a connection
221 SqlConnectionStringBuilder builder = new SqlConnectionStringBuilder(connectionString);
222 if (string.IsNullOrWhiteSpace(builder.DataSource))
223 {
224 Debug.WriteLine("Connection string is missing a server/data source. Please configure a valid SQL Server instance.");
225 return false;
226 }
227
228 // Ensure a short timeout for the validation attempt
229 int originalTimeout = builder.ConnectTimeout;
230 if (originalTimeout <= 0 || originalTimeout > 10)
231 {
232 builder.ConnectTimeout = 5;
233 }
234
235 try
236 {
237 using SqlConnection conn = new SqlConnection(builder.ConnectionString);
238 await conn.OpenAsync().ConfigureAwait(false);
239 await conn.CloseAsync().ConfigureAwait(false);
240 }
241 catch (Exception ex)
242 {
243 Debug.WriteLine($"Connection string validation failed: {ex}");
244 return false;
245 }
246 return true;
247 }
