160 public async Task LoadAsync(Guid id)
161 {
162 // report loading status (BeginLoadingOrSaving should return an IDisposable that clears the loading flag)
163 using (_statusInfoServices.BeginLoadingOrSaving())
164 {
165
166 try
167 {
168 // Call repository inside try so synchronous exceptions are caught here
169 Task<TEntity?> loadTask = Repository.GetByIdAsync(id);
170 TEntity? entity = await loadTask;
171
172 // Set the backing Entity and invoke load hook if found
173 SelectedItem = entity;
174
175 if (entity != null)
176 {
177 await OnLoadFormAsync(entity);
178 IsEditMode = true;
179 }
180 else
181 {
182 IsEditMode = false;
183 }
184 }
185 catch (Exception)
186 {
187 // swallow/log as intended by tests (do not add error message)
188 }
189 finally
190 {
191 // Ensure PropertyChanged for Entity is raised even when repository throws

348 protected void RefreshCommandStates()
349 {
350 // Each RaiseCanExecuteChanged may trigger UI handlers that can throw (for example COMExceptions
351 // from UI interop). Invoke each command's raiser individually and swallow exceptions so that
352 // a single faulty handler cannot break view-model logic. This mirrors the defensive approach
353 // used in other services (e.g. StatusInfoService.NotifyStatusChanged) and in ViewModelBase
354 // where COMExceptions from PropertyChanged handlers are swallowed.
355 try { (AddCommand as RelayCommand)?.RaiseCanExecuteChanged(); } catch (System.Runtime.InteropServices.COMException) { } catch { }
356 try { (SaveCommand as RelayCommand)?.RaiseCanExecuteChanged(); } catch (System.Runtime.InteropServices.COMException) { } catch { }
357 try { (DeleteCommand as RelayCommand)?.RaiseCanExecuteChanged(); } catch (System.Runtime.InteropServices.COMException) { } catch { }
358 try { (CancelCommand as RelayCommand)?.RaiseCanExecuteChanged(); } catch (System.Runtime.InteropServices.COMException) { } catch { }
359 }
