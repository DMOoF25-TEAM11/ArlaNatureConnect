586 {

588 {
589 Person? savedItem = null;

591 {
592 IsSaving = true;
593

596 Address createdAddress = await OnAddAddressFormAsync();
597 Address rs = await _addressRepository!.AddAsync(createdAddress);
598 Debug.Assert(rs != null);
599 Debug.WriteLine("Created Address with Id: " + rs.Id);
600 AddressId = rs.Id;
601

603 await Repository.AddAsync(toAdd);
604 // set base entity so callers can access it
605 SelectedItem = toAdd;
606 savedItem = toAdd;
607 }
608 else
609 {
610 // Update existing entity
611 Person? existing = SelectedItem;
612 if (existing == null)
613 {
614 // nothing to update

617
618 // Use RoleId as the authoritative source when SelectedRole is not available
619 existing.RoleId = SelectedRole?.Id ?? RoleId;
620 existing.AddressId = AddressId;
621 existing.FirstName = FirstName ?? string.Empty;
622 existing.LastName = LastName ?? string.Empty;
623 existing.Email = Email ?? string.Empty;
624 existing.IsActive = IsActive;
625
626 await Repository.UpdateAsync(existing);
627 savedItem = existing;
628 }
629 }

631 {
632 IsSaving = false;
633 }

685 private static object? GetNestedPropertyValue(object? obj, string propertyPath)
686 {
687 if (obj == null || string.IsNullOrWhiteSpace(propertyPath)) return null;
688
689 object? current = obj;
690 Type type = current.GetType();
691 foreach (string part in propertyPath.Split('.'))
692 {
693 PropertyInfo? prop = type.GetProperty(part);
694 if (prop == null) return null;
695 current = prop.GetValue(current, null);
696 if (current == null) return null;
697 type = current.GetType();
698 }
699 // Inline comment: returns the final nested value (could be primitive or object)
700 return current;
701 }

706 private void ApplySearchFilter()
707 {
708 FilteredItems.Clear();
709 if (string.IsNullOrWhiteSpace(SearchText))
710 {
711 foreach (Person person in Items)
712 {
713 FilteredItems.Add(person);
714 }
715 }
716 else
717 {
718 string search = SearchText.Trim().ToLowerInvariant();
719 foreach (Person person in Items)
720 {
721 string firstName = person.FirstName?.ToLowerInvariant() ?? string.Empty;
722 string lastName = person.LastName?.ToLowerInvariant() ?? string.Empty;
723 string email = person.Email?.ToLowerInvariant() ?? string.Empty;
724 string fullName = $"{firstName} {lastName}".Trim();
725
726 if (firstName.Contains(search) ||
727 lastName.Contains(search) ||
728 email.Contains(search) ||
729 fullName.Contains(search))
730 {
731 FilteredItems.Add(person);
732 }
733 }
734 }
